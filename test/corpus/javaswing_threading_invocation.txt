class ThreadingInvocationExample {
  void init(Runnable task) throws Exception {
    SwingUtilities.invokeLater(task);
    javax.swing.SwingUtilities.invokeAndWait(new Runnable() {
      public void run() {
        refresh();
      }
    });
    EventQueue.invokeLater(() -> refresh());
    java.awt.EventQueue.invokeAndWait(task);
  }
}
