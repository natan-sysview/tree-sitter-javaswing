class Example {
  void init(JScrollPane scrollPane, JPanel field, Border border) {
    scrollPane.setViewportBorder(new TitledBorder("Eventos"));
    field.setBorder(null);
    field.setBorder(border);
    this.setBorder(UIManager.getBorder("Panel.border"));
  }
}
