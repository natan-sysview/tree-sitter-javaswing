    JOptionPane.setDefaultLocale(new Locale("es", "MX"));
    JOptionPane.showMessageDialog(null, "Guardado correctamente");
    int answer = JOptionPane.showConfirmDialog(JOptionPane.getFrameForComponent(frame), "Continuar?", "Aviso", JOptionPane.YES_NO_OPTION);

    JOptionPane.showOptionDialog(frame, "Cancelar?", "Aviso", JOptionPane.YES_NO_OPTION,
        JOptionPane.WARNING_MESSAGE, null, options, options[0]);
  }
}
