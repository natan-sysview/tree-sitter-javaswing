Java Swing keyboard bindings
========================================

class KeyboardBindingsExample {
  void init(JComponent component, Action action) {
    KeyStroke enter = KeyStroke.getKeyStroke("ENTER");
    Object key = "submit";
    component.getInputMap(JComponent.WHEN_FOCUSED).put(enter, key);
    component.getActionMap().put(key, action);
    javax.swing.KeyStroke.getKeyStroke("ESCAPE");
  }
}
