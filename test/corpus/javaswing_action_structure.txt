class ActionStructureExample {
  void init(JButton button) {
    Action save = new AbstractAction("Save") {
      public void actionPerformed(ActionEvent event) {
        save();
      }
    };
    button.setAction(new javax.swing.AbstractAction() {
      public void actionPerformed(ActionEvent event) {
        cancel();
      }
    });
  }
}
