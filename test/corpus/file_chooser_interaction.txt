class Example {
  void init(JFileChooser fileChooser, JPanel panel) {
    int open = fileChooser.showOpenDialog(null);
    int save = fileChooser.showSaveDialog(panel);
    int custom = fileChooser.showDialog(panel, "Crear txt");
    if (open == JFileChooser.APPROVE_OPTION) {
      String path = fileChooser.getSelectedFile().getAbsolutePath();
    }
  }
}

class Example {
  void init(JFileChooser fc, FileFilter filter) {
    fc.setFileSelectionMode(JFileChooser.FILES_ONLY);
    fc.setAcceptAllFileFilterUsed(false);
    fc.setFileFilter(filter);
    fc.setApproveButtonText("Archivo Seleccionado");
    fc.setDialogType(JFileChooser.SAVE_DIALOG);
  }
}

========================================
Selected files and multi selection
========================================

class Example {
  void init(JFileChooser chooser) {
    chooser.setMultiSelectionEnabled(true);
    chooser.setSelectedFile(new File("salida.xls"));
    File file = chooser.getSelectedFile();
    File[] files = chooser.getSelectedFiles();
    if (chooser.isMultiSelectionEnabled()) {
      chooser.setSelectedFiles(files);
    }
  }
}

class Example {
  void init(JFileChooser chooser) {
    chooser.setCurrentDirectory(new File("/tmp"));
    chooser.changeToParentDirectory();
    chooser.rescanCurrentDirectory();
    chooser.approveSelection();
    chooser.cancelSelection();
  }
}
