class TransferHandlerInteractionExample {
  void init(JTable table) {
    TransferHandler handler = new TransferHandler("text");
    table.setTransferHandler(handler);
    table.getTransferHandler();
    table.setDragEnabled(true);
    table.setTransferHandler(new javax.swing.TransferHandler("value"));
  }
}
